## 新しく知ったこと
```python
d1 = {'A': 111, 'B': 222, 'C': 333}
d2 = {'D': 444, 'E': 555}
d3 = {'F': 666}
new_dict = {}
for d in [d1, d2, d3]:    # 1回目のループではd1の中身がまるごと出てきてdに格納される。2回目ではd2が、3回目ではd3が。
    for k, v in d.items():    # 外ループで出てきた辞書dからKeyとValueを1ペアずつ取り出し、それぞれ変数ｋと変数ｖに格納
        new_dict[k] = v
print(f'連結した辞書 : {new_dict}')

# 模範解答　update(Key=Value)メソッドを使用、forのネストなし
d1 = {'A': 111, 'B': 222, 'C': 333}
d2 = {'D': 444, 'E': 555}
d3 = {'F': 666}
new_dict = {}
for d in [d1, d2, d3]:
    new_dict.update(d)    # updateメソッドには辞書まるごとも渡せるらしい
print(f'連結した辞書 : {new_dict}')


# 標準の機能を使う方法
d1 = {'A': 111, 'B': 222, 'C': 333}
d2 = {'D': 444, 'E': 555}
d3 = {'F': 666}

new_d = {**d1, **d2, **d3}
print(f'連結した辞書 : {new_d}')

keys = ['A', 'B', 'C']
values = [111, 222 ,333]
d = {keys[0]: values[0], keys[1]: values[1], keys[2]: values[2]}
print(f'作成した辞書 : {d}')

# 模範解答　辞書の内包表記を使用
keys = ['A', 'B', 'C']
values = [111, 222 ,333]
d = {k:v for k, v in zip(keys, values)}
print(f'作成した辞書 : {d}')    # 作成した辞書 : {'A': 111, 'B': 222, 'C': 333}
```
