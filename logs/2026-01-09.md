# 失敗　range()はループ前に作られた数字列を回すだけだから、「最終的に必要な回数」を最初に決めないとだめ！
l = [1, 2, 3, 4, 5]
for i in range(0, len(l), 2):
    l.insert(i, "list")
print(f'"list"を追加したリスト : {l}') # "list"を追加したリスト : ['list', 1, 'list', 2, 'list', 3, 4, 5]

# 再チャレンジ
l = [1, 2, 3, 4, 5]
for i in range(0, len(l)*2, 2):    # 偶数のインデックス位置にlの長さ分だけ要素を追加したら、ループ終了時にはlは「ループ前のlの2倍の長さになってる」から
    l.insert(i, "list")
print(f'"list"を追加したリスト : {l}')    # "list"を追加したリスト : ['list', 1, 'list', 2, 'list', 3, 'list', 4, 'list', 5]

 失敗　for文バージョン
# 設計メモ
# l = [1, 3, 2, 3, 4, 6, 5, 8, 7]を宣言
# lの要素の中から、「要素を3で割ったときの余りが0ではなく、かつ、要素のインデックス番号を3で割ったときの余りも0ではない」要素を選んで新しいリストnew_lに格納
# 「削除後のリスト : 」とnew_lをprintで出力
l = [1, 3, 2, 3, 4, 6, 5, 8, 7]
new_l = []
for i, v in enumerate(l):
    if i % 3 != 0 and v % 3 != 0:
        new_l.append(v)
print(f'削除後のリスト : {new_l}')    # 削除後のリスト : [2, 4, 8, 7]
# ※「インデックスが3の倍数ではなく、かつ、値も3の倍数ではない場合だけ新リストに追加する」設計になってるから、
# 　「インデックスだけが3の倍数」・「値だけが3の倍数」のものまで捨てられてしまってる

# ヒントもらって再チャレンジ
l = [1, 3, 2, 3, 4, 6, 5, 8, 7]
new_l = []
for i, v in enumerate(l):
    if i % 3 == 0 and v % 3 == 0:
        continue
    new_l.append(v)    # これなら、if条件以外の要素は全て新リストに格納できる
print(f'削除後のリスト : {new_l}')    # 削除後のリスト : [1, 3, 2, 4, 6, 5, 8, 7]

# 模範解答　再チャレンジのリスト内包表記バージョン
l = [1, 3, 2, 3, 4, 6, 5, 8, 7]
new_l = [v for i, v in enumerate(l) if not(i % 3 == 0 and v % 3 == 0)]
print(f'削除後のリスト : {new_l}')    # 削除後のリスト : [1, 3, 2, 4, 6, 5, 8, 7]

# 模範解答　「～で始まる要素」を特定するのにスライスを使う方法
telephone_numbers = ['080-1203-4455', '090-9372-9682', '090-3080-4982', '080-3917-5918']
new_telephone_numbers = [n for n in telephone_numbers if "080" in n[:3]]
print(f'080で始まる電話番号 : {new_telephone_numbers}')
